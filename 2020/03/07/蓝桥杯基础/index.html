void allinit(void) // 初始化关闭其他模块(led、继电器、蜂咛器、数码管)
{
  P2=0XA0;          // 继电器，蜂咛器
  P0=0X00;        

  P2=0X80;          // 小灯
  P0=0XFF;            

  P2=0XC0;          // 段选 1~8
  P0=0XFF;

  P2=0XFF;
  P0=0XFF;
}

void display(uchar channel,uchar number)        // 位选 段选         
{
    P2=0XC0;                                    // 段选
    P0=0X01<<channel;

    P2=0XFF;                                    // 段选
    P0=tab[number];
    delayms(1);
}
