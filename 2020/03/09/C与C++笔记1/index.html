15.冒泡排序法

void fun(int c[], int d)
{
    int temp=0;
    for (int a = 0; a < d - 1; a++)
    {
        for (int b = 0; b < d - 1 - a; b++)
        {
            if (c[b] <c[b + 1])
            {
                temp = c[b];
                c[b] = c[b + 1];
                c[b + 1] = temp;
            }
        }
    }
}
